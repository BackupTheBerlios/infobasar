Vergleich der Wiki-Syntax:

Leider gibt es so viele Syntaxvarianten, wie es Wiki-Systeme gibt. Ich habe mich in MoinMoin und PHPWiki informiert und versucht, daraus die besten Ideen zusammenzufassen. Ziel war eine Syntax, die mit wenig Regeln, die selber einfach zu formulieren sind, auskommt. Damit sollte sie leicht zu erlernen sein.

function getMicroTime($time = null){
  if (empty ($time))
    $time = microtime ();
  list($usec, $sec) = explode(" ", $time);
  return ((float) $usec + (float)$sec);
}
function getRuntimeString (){
  global $hm_start_time;
  $time = getMicroTime () - getMicroTime ($hm_start_time);
  $time = substr ($time, 0, strpos ($time, '.') + 7);
  return  "<br/>Dauer: $time sec";
}
